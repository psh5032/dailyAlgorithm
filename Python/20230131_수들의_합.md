# 나의 풀이
```python
import sys
sys.stdin=open("input.txt","rt")
n,m = map(int, input().split())
a=list(map(int, input().split()))
lt=0
rt=1
tot=a[0]
cnt=0

while True:
    if tot<m:
        if rt==n:
            break
        tot+=a[rt]
        rt+=1
    elif tot==m:
        cnt+=1
        tot-=a[lt]
        lt+=1
    elif tot>m: # else로 작성하기
        tot-=a[lt]
        lt+=1
print(cnt)
```
